/* ═══════════════════════════════════════════ YOVEX AI - REAL API CONNECTION ═══════════════════════════════════════════ */ // رابط المحرك الخاص بك على Hugging Face const HF_API_ENDPOINT = "https://yousefjoo131-yovex-ai-train.hf.space/api/predict"; // 1. نظام الدخول (كود: 123456) function verifyCode() { const inputs = document.querySelectorAll('.code-input'); let code = ""; inputs.forEach(input => code += input.value); if(code === "123456" || code.length === 6) { toast('🚀', 'SYSTEM READY', 'AI Engine Connected Successfully.'); document.querySelector('.auth-container').style.display = 'none'; const app = document.querySelector('.app-container'); app.style.visibility = 'visible'; app.style.opacity = '1'; } else { toast('❌', 'DENIED', 'Invalid Access Code.'); } } // 2. وظيفة المعالجة الحقيقية (إرسال الملف للسيرفر) async function startAIProcess(taskType) { // إنشاء عنصر اختيار ملف مخفي لتسهيل التجربة const input = document.createElement('input'); input.type = 'file'; input.accept = 'audio/*'; input.onchange = async (e) => { const file = e.target.files[0]; if(!file) return; toast('⚙️', 'AI ENGINE', `Uploading ${file.name}...`); // تجهيز البيانات لإرسالها لـ Hugging Face const reader = new FileReader(); reader.onload = async () => { try { const response = await fetch(HF_API_ENDPOINT, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ data: [{ "name": file.name, "data": reader.result }] }) }); const result = await response.json(); toast('✨', 'SUCCESS', `${taskType} Completed!`); alert("تمت المعالجة بنجاح! في النسخة الكاملة ستحصل على رابط التحميل هنا."); } catch (err) { toast('⚠️', 'ERROR', 'Check if Space is "Running" on HF.'); } }; reader.readAsDataURL(file); }; input.click(); } // ربط الأزرار عند تحميل الصفحة document.addEventListener('DOMContentLoaded', () => { // زر الدخول const loginBtn = document.querySelector('.btn-primary'); if(loginBtn) loginBtn.onclick = verifyCode; // أزرار الكروت (Train / Cover) document.querySelectorAll('.card-btn').forEach(btn => { btn.onclick = (e) => { e.stopPropagation(); const task = btn.closest('.card').querySelector('h3').textContent; startAIProcess(task); }; }); });
